 ### npm рЃърЃљрЃЎрЃћрЃбрЃў

 ## рЃЎрЃљрЃЋрЃерЃўрЃарЃћрЃЉрЃў
 рЃЏрЃЮрЃЉрЃўрЃџрЃБрЃарЃўрЃА рЃДрЃЋрЃћрЃџрЃљ рЃАрЃљрЃфрЃљрЃЋрЃерЃў рЃгрЃЋрЃЊрЃЮрЃЏрЃљ рЃ«рЃЊрЃћрЃЉрЃљ рЃЎрЃљрЃЋрЃерЃўрЃарЃўрЃА рЃАрЃљрЃерЃБрЃљрЃџрЃћрЃЉрЃўрЃЌ.

 Mobiletto рЃљрЃЏрЃЪрЃљрЃЏрЃљрЃЊ рЃЏрЃ«рЃљрЃарЃА рЃБрЃГрЃћрЃарЃА рЃЎрЃљрЃЋрЃерЃўрЃарЃћрЃЉрЃА рЃљрЃЊрЃњрЃўрЃџрЃЮрЃЉрЃарЃўрЃЋ рЃцрЃљрЃўрЃџрЃБрЃа рЃАрЃўрЃАрЃбрЃћрЃЏрЃљрЃАрЃЌрЃљрЃю рЃЊрЃљ Amazon S3 рЃЌрЃљрЃўрЃњрЃБрЃџрЃћрЃЉрЃЌрЃљрЃю.

 рЃЎрЃљрЃЋрЃерЃўрЃарЃћрЃЉрЃў рЃЏрЃўрЃЌрЃўрЃЌрЃћрЃЉрЃБрЃџрЃўрЃљ JSON рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃўрЃЌ, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃљрЃАрЃћ рЃњрЃљрЃЏрЃЮрЃўрЃДрЃБрЃарЃћрЃЉрЃљ:

    {
        "name": "name-of-connection",
        "type": "s3", # one of "s3", "b2", or "local"
        "opts": {
            "readOnly": false, # optional, default false
            "cacheSize": 100, # optional,
            ... other type-specific options ...
        },
        "encryption": { # optional, default is no encryption
            "key": "enc-key", # required if encryption is enabled
            "iv": "enc-IV", # optional, derived from key if omitted
            "algo": "enc-algo" # optional, default is aes-256-cbc
        }
    }

 рЃДрЃЋрЃћрЃџрЃљ рЃЋрЃљрЃарЃўрЃљрЃюрЃбрЃўрЃА рЃЊрЃљ рЃЏрЃљрЃЌрЃў рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃћрЃЉрЃўрЃАрЃљ рЃЊрЃљ рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЉрЃћрЃЉрЃўрЃА рЃерЃћрЃАрЃљрЃ«рЃћрЃЉ рЃАрЃарЃБрЃџрЃў рЃЊрЃћрЃбрЃљрЃџрЃћрЃЉрЃўрЃАрЃЌрЃЋрЃўрЃА рЃўрЃ«рЃўрЃџрЃћрЃЌ
 [mobiletto docs](https://www.npmjs.com/package/mobiletto#Basic-usage).

 ### рЃерЃћрЃЦрЃЏрЃћрЃюрЃўрЃЌ рЃЎрЃљрЃЋрЃерЃўрЃарЃў
 рЃљрЃ«рЃљрЃџрЃў рЃЎрЃљрЃЋрЃерЃўрЃарЃўрЃА рЃерЃћрЃАрЃљрЃЦрЃЏрЃюрЃћрЃџрЃљрЃЊ, рЃЕрЃЋрЃћрЃю рЃњрЃЋрЃГрЃўрЃарЃЊрЃћрЃЉрЃљ рЃћрЃарЃЌ-рЃћрЃарЃЌрЃў JSON рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃў.

 рЃЕрЃЋрЃћрЃю рЃерЃћрЃњрЃЋрЃўрЃФрЃџрЃўрЃљ рЃерЃћрЃЋрЃЦрЃЏрЃюрЃљрЃЌ рЃћрЃарЃЌрЃў рЃўрЃюрЃбрЃћрЃарЃљрЃЦрЃбрЃўрЃБрЃџрЃљрЃЊ, рЃљрЃю рЃЏрЃўрЃЋрЃљрЃгрЃЮрЃЊрЃЮрЃЌ рЃћрЃарЃЌрЃў рЃцрЃљрЃўрЃџрЃўрЃА рЃљрЃю JSON рЃџрЃўрЃбрЃћрЃарЃљрЃџрЃБрЃарЃљрЃЊ:

    # Create JSON via interactive prompts
    mo connect

    # Create via interactive prompts, but force name `my-conn` to be used
    mo connect -c my-conn
    mo connect --create my-conn # long-form option

    # Create from JSON config
    mo connect my-conn.json

    # Create from JSON config, but force name `renamed` to be used
    mo connect -c renamed my-conn.json

    # Create from literal JSON object
    mo connect '{
        "name": "my-conn",
        "type": "local",
        "key": "/tmp/basedir"
    }'

 ### рЃЕрЃљрЃЏрЃЮрЃЌрЃЋрЃљрЃџрЃћрЃЌ рЃЎрЃљрЃЋрЃерЃўрЃарЃћрЃЉрЃў
 рЃЕрЃљрЃЏрЃЮрЃЌрЃЋрЃљрЃџрЃћрЃЌ рЃ«рЃћрЃџрЃЏрЃўрЃАрЃљрЃгрЃЋрЃЊрЃЮрЃЏрЃў рЃЎрЃљрЃЋрЃерЃўрЃарЃћрЃЉрЃў:

    mo connect -l
    mo connect --list

 ### рЃњрЃљрЃЊрЃљрЃДрЃљрЃарЃћрЃЌ рЃЎрЃљрЃЋрЃерЃўрЃарЃў JSON рЃЎрЃЮрЃюрЃцрЃўрЃњрЃБрЃарЃљрЃфрЃўрЃљ
 рЃљрЃЕрЃЋрЃћрЃюрЃћрЃЌ рЃАрЃарЃБрЃџрЃў JSON рЃЎрЃЮрЃюрЃцрЃўрЃњрЃБрЃарЃљрЃфрЃўрЃљ рЃЎрЃљрЃЋрЃерЃўрЃарЃўрЃАрЃЌрЃЋрЃўрЃА. **рЃњрЃљрЃцрЃарЃЌрЃ«рЃўрЃџрЃћрЃЉрЃљ** рЃћрЃА рЃЊрЃљрЃЉрЃћрЃГрЃЊрЃљрЃЋрЃА рЃњрЃљрЃАрЃљрЃдрЃћрЃЉрЃћрЃЉрЃА рЃЊрЃљ рЃАрЃљрЃўрЃЊрЃБрЃЏрЃџрЃЮрЃћрЃЉрЃћрЃЉрЃА stdout-рЃўрЃАрЃЌрЃЋрЃўрЃА.

    mo connect -d my-conn
    mo connect --dump my-conn

    # pretty-print JSON
    mo connect --verbose --dump my-conn
    mo connect -vd my-conn

 ### рЃЎрЃљрЃЋрЃерЃўрЃарЃўрЃА рЃгрЃљрЃерЃџрЃљ
 рЃљрЃЏрЃЮрЃўрЃдрЃћрЃЌ рЃћрЃарЃЌрЃў рЃЎрЃљрЃЋрЃерЃўрЃарЃў:

    mo connect -r my-conn
    mo connect --remove my-conn

 ### рЃгрЃљрЃерЃљрЃџрЃћрЃЌ рЃДрЃЋрЃћрЃџрЃљ рЃЎрЃљрЃЋрЃерЃўрЃарЃў
 рЃДрЃЋрЃћрЃџрЃљ рЃЎрЃљрЃЋрЃерЃўрЃарЃўрЃА рЃгрЃљрЃерЃџрЃљ:

    mo connect -R
    mo connect --remove-all

 ## рЃАрЃљрЃфрЃљрЃЋрЃЌрЃљрЃю рЃЏрЃБрЃерЃљрЃЮрЃЉрЃљ
 рЃарЃљрЃЏрЃЊрЃћрЃюрЃўрЃЏрЃћ рЃЎрЃљрЃЋрЃерЃўрЃарЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃўрЃА рЃерЃћрЃЏрЃЊрЃћрЃњ рЃњрЃљрЃЏрЃЮрЃўрЃДрЃћрЃюрЃћрЃЌ `ls` , `cp` , `rm` , `meta` рЃЊрЃљ `mirror` рЃЮрЃърЃћрЃарЃљрЃфрЃўрЃћрЃЉрЃў
 рЃЏрЃљрЃЌрЃЌрЃљрЃю рЃЏрЃБрЃерЃљрЃЮрЃЉрЃљ.

 рЃДрЃЋрЃћрЃџрЃљ рЃцрЃљрЃўрЃџрЃўрЃА рЃњрЃќрЃљрЃА, рЃарЃЮрЃЏрЃћрЃџрЃАрЃљрЃф рЃЌрЃЦрЃЋрЃћрЃю рЃљрЃгрЃЋрЃЊрЃўрЃЌ рЃЉрЃарЃФрЃљрЃюрЃћрЃЉрЃљрЃА, рЃБрЃюрЃЊрЃљ рЃўрЃДрЃЮрЃА рЃЊрЃљрЃбрЃљрЃюрЃўрЃџрЃў рЃЎрЃљрЃЋрЃерЃўрЃарЃўрЃА рЃАрЃљрЃ«рЃћрЃџрЃўрЃЌ
 рЃћрЃА рЃћрЃ«рЃћрЃЉрЃљ. рЃЦрЃЋрЃћрЃЏрЃЮрЃЌ рЃЏрЃЮрЃфрЃћрЃЏрЃБрЃџ рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃћрЃЉрЃерЃў, рЃЕрЃЋрЃћрЃю рЃЋрЃЋрЃљрЃарЃљрЃБрЃЊрЃЮрЃЉрЃЌ, рЃарЃЮрЃЏ рЃљрЃарЃАрЃћрЃЉрЃЮрЃЉрЃА рЃЮрЃарЃў рЃЎрЃљрЃЋрЃерЃўрЃарЃў рЃАрЃљрЃ«рЃћрЃџрЃљрЃЊ `conn1` рЃЊрЃљ `conn2`

 ### рЃЕрЃљрЃЏрЃЮрЃЌрЃЋрЃљрЃџрЃћрЃЌ рЃцрЃљрЃўрЃџрЃћрЃЉрЃў
 рЃАрЃўрЃўрЃА рЃЉрЃарЃФрЃљрЃюрЃћрЃЉрЃўрЃА рЃњрЃљрЃЏрЃЮрЃЏрЃљрЃЋрЃљрЃџрЃў рЃљрЃарЃўрЃА JSON рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃўрЃА рЃАрЃћрЃарЃўрЃљ, рЃЌрЃўрЃЌрЃЮ рЃАрЃбрЃарЃўрЃЦрЃЮрЃюрЃќрЃћ, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃгрЃљрЃарЃЏрЃЮрЃљрЃЊрЃњрЃћрЃюрЃА рЃцрЃљрЃўрЃџрЃћрЃЉрЃА
 рЃюрЃљрЃърЃЮрЃЋрЃюрЃўрЃљ. рЃљрЃЏ рЃЮрЃЉрЃўрЃћрЃЦрЃбрЃћрЃЉрЃА рЃћрЃЦрЃюрЃћрЃЉрЃљ `name` рЃЊрЃљ `type` " рЃЊрЃљ рЃерЃћрЃАрЃљрЃФрЃџрЃЮрЃљ рЃАрЃ«рЃЋрЃљ рЃЋрЃћрЃџрЃћрЃЉрЃў.

    mo ls conn1/some/path

 рЃЕрЃљрЃЏрЃЮрЃЌрЃЋрЃљрЃџрЃћрЃЌ рЃцрЃљрЃўрЃџрЃћрЃЉрЃў рЃарЃћрЃЎрЃБрЃарЃАрЃўрЃБрЃџрЃљрЃЊ:

    mo ls -r conn1/some/path
    mo ls --recursive conn1/some/path

 ** рЃЊрЃљрЃерЃўрЃцрЃарЃБрЃџрЃў рЃерЃћрЃюрЃљрЃ«рЃЋрЃўрЃА рЃерЃћрЃюрЃўрЃерЃЋрЃюрЃљ **: рЃарЃћрЃЎрЃБрЃарЃАрЃўрЃБрЃџрЃў рЃЕрЃљрЃЏрЃЮрЃюрЃљрЃЌрЃЋрЃљрЃџрЃў рЃ»рЃћрЃа рЃљрЃа рЃљрЃарЃўрЃА рЃЏрЃ«рЃљрЃарЃЊрЃљрЃГрЃћрЃарЃўрЃџрЃў рЃЊрЃљрЃерЃўрЃцрЃарЃБрЃџрЃў рЃЎрЃљрЃЋрЃерЃўрЃарЃћрЃЉрЃўрЃАрЃЌрЃЋрЃўрЃА.
 рЃЌрЃЦрЃЋрЃћрЃю рЃерЃћрЃњрЃўрЃФрЃџрЃўрЃљрЃЌ рЃЕрЃљрЃЏрЃЮрЃЌрЃЋрЃљрЃџрЃЮрЃЌ рЃцрЃљрЃўрЃџрЃћрЃЉрЃўрЃА рЃЏрЃ«рЃЮрЃџрЃЮрЃЊ рЃћрЃарЃЌрЃў рЃЊрЃўрЃарЃћрЃЦрЃбрЃЮрЃарЃўрЃљ рЃћрЃарЃЌрЃЊрЃарЃЮрЃБрЃџрЃљрЃЊ.

 рЃЕрЃљрЃЏрЃЮрЃЌрЃЋрЃљрЃџрЃћрЃЌ рЃцрЃљрЃўрЃџрЃћрЃЉрЃў рЃАрЃўрЃбрЃДрЃЋрЃўрЃћрЃарЃљрЃЊ (рЃАрЃљрЃЎрЃЏрЃљрЃЮрЃЊ рЃЊрЃљрЃЉрЃћрЃГрЃЊрЃўрЃџрЃў JSON):

    mo ls -v conn1/some/path
    mo ls --verbose conn1/some/path

 рЃЕрЃљрЃЏрЃЮрЃЌрЃЋрЃљрЃџрЃћрЃЌ рЃцрЃљрЃўрЃџрЃћрЃЉрЃў рЃарЃћрЃЎрЃБрЃарЃАрЃўрЃБрЃџрЃљрЃЊ рЃЊрЃљ рЃАрЃўрЃбрЃДрЃЋрЃўрЃћрЃарЃљрЃЊ:

    mo ls -vr conn1/some/path
    mo ls --verbose --recursive conn1/some/path

 рЃЊрЃљрЃ«рЃЏрЃљрЃарЃћрЃЉрЃўрЃА рЃюрЃљрЃ«рЃЋрЃљ `ls` рЃЉрЃарЃФрЃљрЃюрЃћрЃЉрЃўрЃАрЃЌрЃЋрЃўрЃА, рЃљрЃдрЃгрЃћрЃарЃА рЃДрЃЋрЃћрЃџрЃљ рЃЋрЃљрЃарЃўрЃљрЃюрЃбрЃА:

    mo ls -h
    mo ls --help

 ### рЃЕрЃљрЃгрЃћрЃарЃћрЃЌ рЃцрЃљрЃўрЃџрЃў stdout-рЃўрЃАрЃЌрЃЋрЃўрЃА
 `cat` рЃЉрЃарЃФрЃљрЃюрЃћрЃЉрЃљ рЃЊрЃљрЃЉрЃћрЃГрЃЊрЃљрЃЋрЃА рЃцрЃљрЃўрЃџрЃА stdout-рЃерЃў:

    mo cat conn1/some/file # print to terminal
    mo cat conn1/some/file | some-other-command # use in command pipeline

 UNIX-рЃўрЃА рЃГрЃћрЃерЃЏрЃљрЃарЃўрЃбрЃў рЃАрЃБрЃџрЃўрЃАрЃЎрЃЋрЃћрЃЌрЃћрЃЉрЃўрЃЌ *(рЃЌрЃБрЃЏрЃфрЃљ рЃарЃљ рЃЌрЃЦрЃЏрЃљ рЃБрЃюрЃЊрЃљ рЃљрЃарЃљ рЃърЃарЃљрЃЦрЃбрЃўрЃЎрЃљрЃерЃў!)* mobiletto-рЃА `cat` рЃЉрЃарЃФрЃљрЃюрЃћрЃЉрЃљ **рЃљрЃа рЃўрЃдрЃћрЃЉрЃА рЃЋрЃљрЃарЃўрЃљрЃюрЃбрЃћрЃЉрЃА**
 рЃњрЃљрЃарЃЊрЃљ `-h` / `--help` рЃАрЃљрЃАрЃљрЃарЃњрЃћрЃЉрЃџрЃЮ рЃўрЃюрЃцрЃЮрЃарЃЏрЃљрЃфрЃўрЃўрЃА рЃЕрЃЋрЃћрЃюрЃћрЃЉрЃўрЃАрЃЌрЃЋрЃўрЃА.

 ### рЃЊрЃљрЃљрЃЎрЃЮрЃърЃўрЃарЃћрЃЌ рЃцрЃљрЃўрЃџрЃћрЃЉрЃў
 рЃЊрЃљрЃљрЃЎрЃЮрЃърЃўрЃарЃћрЃЌ рЃћрЃарЃЌрЃў рЃцрЃљрЃўрЃџрЃў рЃћрЃарЃЌрЃў рЃЏрЃЮрЃЉрЃўрЃџрЃБрЃарЃўрЃЊрЃљрЃю рЃЏрЃћрЃЮрЃарЃћрЃќрЃћ:

    mo cp conn1/some/path/to/file conn2/some/dest/path/

 рЃЊрЃљрЃљрЃЎрЃЮрЃърЃўрЃарЃћрЃЌ рЃћрЃарЃЌрЃў рЃцрЃљрЃўрЃџрЃў рЃћрЃарЃЌрЃў рЃЏрЃЮрЃЉрЃўрЃџрЃБрЃарЃўрЃЊрЃљрЃю рЃЏрЃћрЃЮрЃарЃћрЃќрЃћ, рЃњрЃљрЃЊрЃљрЃљрЃарЃЦрЃЋрЃћрЃЌ рЃАрЃљрЃ«рЃћрЃџрЃў рЃЊрЃљрЃюрЃўрЃерЃюрЃБрЃџрЃћрЃЉрЃўрЃА рЃљрЃЊрЃњрЃўрЃџрЃќрЃћ:

    mo cp conn1/some/path/to/file conn2/some/dest/path/file2

 рЃарЃћрЃЎрЃБрЃарЃАрЃўрЃБрЃџрЃљрЃЊ рЃЊрЃљрЃљрЃЎрЃЮрЃърЃўрЃарЃћрЃЌ рЃЊрЃўрЃарЃћрЃЦрЃбрЃЮрЃарЃўрЃљ:

    mo cp -r conn1/some/directory conn2/some/dest/

 рЃЊрЃљрЃ«рЃЏрЃљрЃарЃћрЃЉрЃўрЃА рЃюрЃљрЃ«рЃЋрЃљ `cp` рЃЉрЃарЃФрЃљрЃюрЃћрЃЉрЃўрЃАрЃЌрЃЋрЃўрЃА, рЃљрЃдрЃгрЃћрЃарЃА рЃДрЃЋрЃћрЃџрЃљ рЃЋрЃљрЃарЃўрЃљрЃюрЃбрЃА:

    mo cp -h
    mo cp --help

 рЃерЃћрЃюрЃўрЃерЃЋрЃюрЃљ: рЃцрЃљрЃўрЃџрЃўрЃА рЃЎрЃЮрЃърЃўрЃарЃћрЃЉрЃљ рЃљрЃа рЃљрЃарЃўрЃА рЃФрЃљрЃџрЃўрЃљрЃю рЃАрЃгрЃарЃљрЃцрЃў, рЃарЃљрЃЊрЃњрЃљрЃю рЃўрЃА рЃљрЃЏрЃЪрЃљрЃЏрЃљрЃЊ рЃАрЃљрЃГрЃўрЃарЃЮрЃћрЃЉрЃА рЃерЃБрЃљрЃЏрЃљрЃЋрЃљрЃџ рЃбрЃћрЃЏрЃъ рЃцрЃљрЃўрЃџрЃА.

 ### р▓грЃљрЃерЃљрЃџрЃћ рЃцрЃљрЃўрЃџрЃћрЃЉрЃў
 рЃгрЃљрЃерЃљрЃџрЃћрЃЌ рЃћрЃарЃЌрЃў рЃцрЃљрЃўрЃџрЃў:

    mo rm conn1/some/file.txt

 рЃгрЃљрЃерЃљрЃџрЃћрЃЌ рЃћрЃарЃЌрЃў рЃцрЃљрЃўрЃџрЃў рЃЊрЃљ рЃюрЃБ рЃБрЃЕрЃўрЃЋрЃўрЃЌ рЃерЃћрЃфрЃЊрЃЮрЃЏрЃћрЃЉрЃА:

    mo rm -f conn1/some/file.txt
    mo rm --force conn1/some/file.txt

 рЃгрЃљрЃерЃљрЃџрЃћ рЃЊрЃўрЃарЃћрЃЦрЃбрЃЮрЃарЃўрЃљ:

    mo rm -r conn1/some/directory
    mo rm --recursive conn1/some/directory

 рЃгрЃљрЃерЃљрЃџрЃћ рЃЊрЃўрЃарЃћрЃЦрЃбрЃЮрЃарЃўрЃљ рЃЊрЃљ рЃљрЃа рЃБрЃЕрЃўрЃЮрЃЊрЃћ рЃерЃћрЃфрЃЊрЃЮрЃЏрЃћрЃЉрЃА:

    mo rm -rf conn1/some/directory
    mo rm --recursive --force conn1/some/directory

 ### рЃЏрЃћрЃбрЃљрЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃћрЃЉрЃўрЃА рЃюрЃљрЃ«рЃЋрЃљ
 рЃ«рЃерЃўрЃарЃљрЃЊ рЃАрЃљрЃАрЃљрЃарЃњрЃћрЃЉрЃџрЃЮрЃљ рЃцрЃљрЃўрЃџрЃўрЃА рЃЏрЃћрЃбрЃљрЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃћрЃЉрЃўрЃА рЃфрЃЮрЃЊрЃюрЃљ рЃЏрЃЌрЃћрЃџрЃў рЃцрЃљрЃўрЃџрЃўрЃА рЃЕрЃљрЃЏрЃЮрЃбрЃЋрЃўрЃарЃЌрЃЋрЃўрЃА рЃњрЃљрЃарЃћрЃерЃћ.

 Mobiletto-рЃА рЃерЃћрЃБрЃФрЃџрЃўрЃљ рЃЏрЃўрЃБрЃЌрЃўрЃЌрЃЮрЃА рЃцрЃљрЃўрЃџрЃўрЃА рЃќрЃЮрЃЏрЃљ рЃЊрЃљ рЃЉрЃЮрЃџрЃЮ рЃерЃћрЃфрЃЋрЃџрЃўрЃџрЃў рЃЊрЃарЃЮ, рЃњрЃљрЃарЃЊрЃљ рЃАрЃљрЃ«рЃћрЃџрЃўрЃАрЃљ рЃЊрЃљ рЃбрЃўрЃърЃўрЃАрЃљ.

 рЃЏрЃћрЃбрЃљрЃЏрЃЮрЃюрЃљрЃфрЃћрЃЏрЃћрЃЉрЃўрЃА рЃАрЃљрЃюрЃљрЃ«рЃљрЃЋрЃљрЃЊ:

    mo meta conn1/some/file.txt

 ### рЃАрЃљрЃарЃЎрЃўрЃА рЃЊрЃўрЃарЃћрЃЦрЃбрЃЮрЃарЃўрЃћрЃЉрЃў
 Mirroring рЃБрЃцрЃарЃЮ рЃЏрЃћрЃбрЃўрЃљ, рЃЋрЃўрЃЊрЃарЃћ `cp` рЃўрЃА рЃњрЃљрЃюрЃАрЃљрЃЎрЃБрЃЌрЃарЃћрЃЉрЃБрЃџрЃў рЃерЃћрЃЏрЃЌрЃ«рЃЋрЃћрЃЋрЃљ, рЃћрЃА рЃљрЃарЃўрЃА рЃфрЃљрЃџрЃЎрЃћ рЃЏрЃЮрЃЉрЃўрЃџрЃБрЃарЃў рЃЉрЃарЃФрЃљрЃюрЃћрЃЉрЃљ, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃљрЃ«рЃЋрЃћрЃЋрЃА
 рЃЎрЃЮрЃърЃўрЃарЃћрЃЉрЃўрЃА рЃЮрЃърЃћрЃарЃљрЃфрЃўрЃљ; рЃўрЃА рЃњрЃарЃФрЃћрЃџрЃЊрЃћрЃЉрЃљ рЃЏрЃљрЃерЃўрЃюрЃљрЃф рЃЎрЃў, рЃарЃЮрЃЊрЃћрЃАрЃљрЃф рЃ«рЃЊрЃћрЃЉрЃљ рЃерЃћрЃфрЃЊрЃЮрЃЏрЃћрЃЉрЃў рЃЊрЃљ рЃЌрЃЋрЃљрЃџрЃДрЃБрЃарЃў рЃљрЃЊрЃћрЃЋрЃюрЃћрЃЌ рЃгрЃљрЃарЃЏрЃљрЃбрЃћрЃЉрЃћрЃЉрЃўрЃАрЃљ рЃЊрЃљ рЃерЃћрЃфрЃЊрЃЮрЃЏрЃћрЃЉрЃўрЃА рЃарЃљрЃЮрЃЊрЃћрЃюрЃЮрЃЉрЃљрЃА.

 рЃДрЃЋрЃћрЃџрЃљрЃцрЃарЃўрЃА рЃљрЃАрЃљрЃ«рЃЋрЃљ conn1-рЃЊрЃљрЃю conn2-рЃЏрЃЊрЃћ:

    mo mirror conn1 conn2

 рЃЦрЃЋрЃћрЃЊрЃўрЃарЃћрЃЦрЃбрЃЮрЃарЃўрЃА рЃљрЃАрЃљрЃ«рЃЋрЃљ:

    mo mirror conn1/some/subdir conn2

 рЃЦрЃЋрЃћрЃЊрЃўрЃарЃћрЃЦрЃбрЃЮрЃарЃўрЃўрЃА рЃАрЃ«рЃЋрЃљ рЃЦрЃЋрЃћрЃЊрЃўрЃарЃћрЃЦрЃбрЃЮрЃарЃўрЃљрЃерЃў рЃљрЃАрЃљрЃ«рЃЋрЃљ:

    mo mirror conn1/some/subdir conn2/another/dir
